3. IPC: Program handling PIPE

a3.c

// a3.c
#include<stdio.h>
#include<unistd.h>
#include<fcntl.h>
#include<stdlib.h>

int main(){
    int fd1,fd2[2],pid,n;
    char str1[512],str2[512];
    fd1=open("text1.txt",O_RDWR | O_CREAT,0777);
    n=read(fd1,str1,512);
    pipe(fd2);
    pid=fork();

    if(pid>0){
        close(fd2[0]);
        write(fd2[1],str1,n);
    }
    else if(pid==0){
        close(fd2[1]);
        read(fd2[0],str2,512);
        str2[n]='\0';
        puts(str2);
    }
    else{
        printf("Error in fork()");
        exit(0);
    }
    return 0;
}

4. IPC: Program for Client-Server Communication using TCP Socket

Server Side (server4.c)

#include <stdio.h>
#include <string.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#define SERVER_IP "127.0.0.1"
#define SERVER_PORT 4560

void main() {
    struct sockaddr_in client, server;
    int sd, nsd, clen = sizeof(client);
    char msg[512];
    bzero((char *)&server, sizeof(server));
    server.sin_family = AF_INET;
    server.sin_addr.s_addr = inet_addr(SERVER_IP);
    server.sin_port = htons(SERVER_PORT);
    
    sd = socket(AF_INET, SOCK_STREAM, 0);
    bind(sd, (struct sockaddr *)&server, sizeof(server));
    listen(sd, 5);
    
    while(1) {
        nsd = accept(sd, (struct sockaddr *)&client, &clen);
        do {
            memset(msg, 0x0, 512);
            recv(nsd, msg, 512, 0);
            printf("Message received from client : %s\n", msg);
            printf("Client's IP : %s \n", inet_ntoa(client.sin_addr));
            printf("Client's Port : %u\n", ntohs(client.sin_port));
        } while(strcmp(msg, "stop"));
        close(nsd);
    }
}

Server Side (server6.c)

#include <string.h>
#include <sys/types.h>
#include <sys/socket.h>
#include <netinet/in.h>

#define SERVER_PORT 4560

    struct sockaddr_in client, server;
    int sd, nsd, clen = sizeof(client);
    char msg[512], msg1[512] = "Invalid request ";
    char *ptr;
    time_t ti;

                send(nsd, msg1, strlen(msg1)+1, 0);

9. Concurrent ECHO Server using TCP

Server Side (server9.c)

#include<stdio.h>
#include<unistd.h>
#include<sys/socket.h>
#include<sys/types.h>
#include<netinet/in.h>
#include<string.h>
#include<arpa/inet.h>

#define server_ip "127.0.0.3"
#define server_port 5780

int main(){
    struct sockaddr_in client,server;
    int sd,nsd ,clen=sizeof(client);
    char str[512];
    int pid;
    
    bzero((char *)&server,sizeof(server));
    server.sin_addr.s_addr=inet_addr(server_ip);
    server.sin_port=htons(server_port);
    server.sin_family=AF_INET;
    
    sd=socket(AF_INET,SOCK_STREAM,0);
    bind(sd,(struct sockaddr*)&server,sizeof(server));
    listen(sd,5);
    
    while(1)
    {
        nsd=accept(sd,(struct sockaddr*)&client,&clen);
        pid=fork();
        
        if(pid>0){
            close(nsd);
        }
        else if(pid<0){
            printf("error in fork");
            break;
        }
        else{
            do{
                memset(str,0x0,512);
                recv(nsd,str,512,0);
                send(nsd,str,strlen(str)+1,0);
            } while(strcmp(str,"stop"));
        }
    }
}

10. Sliding Window Protocol

Server Side (server10.c)

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <arpa/inet.h>
#include <netinet/in.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <time.h>

#define S_IP "127.0.0.1"
#define S_PORT 15550

int main() {
    struct sockaddr_in client, server;
    int s, ns, clen=sizeof(client);
    char frame;
    int ack=0;
    char msg1[]="Invalid request";
    char msg[512];
    time_t ti;
    
    bzero((char*)&server, sizeof(server));
    server.sin_family=AF_INET;
    server.sin_addr.s_addr=inet_addr(S_IP);
    server.sin_port=htons(S_PORT);
    
    s=socket(AF_INET, SOCK_STREAM, 0);
    bind(s, (struct sockaddr *)&server, sizeof(server));
    listen(s, 5);
    
    ns=accept(s, (struct sockaddr *)&client, &clen);
    
    while(1) {
        int n = recv(ns, &frame, sizeof(char), 0);
        if (n <= 0) break;
        
        printf("received frame: %c\n", frame);
        ack++;
        send(ns, &ack, sizeof(int), 0);
    }
    
    close(ns);
    close(s);
    return 0;
}

Client Side (client10.c)

#include <stdio.h>
#include <string.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>

#define C_IP "127.0.0.1"
#define S_IP "127.0.0.1"
#define S_PORT 15550
#define C_PORT 15550

int main() {
    struct sockaddr_in client, server;
    int s;
    int counter=0, w=3, ack, i;
    char msg[]="MCKVIE";
    int total=strlen(msg);
    
    bzero((char*)&client, sizeof(client));
    client.sin_family=AF_INET;
    client.sin_addr.s_addr=inet_addr(C_IP);
    client.sin_port=htons(C_PORT);
    
    bzero((char*)&server, sizeof(server));
    server.sin_family=AF_INET;
    server.sin_addr.s_addr=inet_addr(S_IP);
    server.sin_port=htons(S_PORT);
    
    s=socket(AF_INET, SOCK_STREAM, 0);
    bind(s, (struct sockaddr *)&client, sizeof(client));
    connect(s, (struct sockaddr *)&server, sizeof(server));
    
    while(counter<total) {
        for(i=counter; i<w+counter && i<total; i++) {
            printf("sent frame %c\n", msg[i]);
            send(s, &msg[i], sizeof(char), 0);
        }
        
        recv(s, &ack, sizeof(int), 0);
        printf("Received ack: %d\n", ack);
        counter=ack+2; 
    }
    
    close(s);
    return 0;
}
