DELIMITER //

-- 1. Factorial
CREATE PROCEDURE GetFactorial(IN num INT, OUT fact BIGINT)
BEGIN
    DECLARE i INT DEFAULT 1;
    SET fact = 1;
    WHILE i <= num DO
        SET fact = fact * i;
        SET i = i + 1;
    END WHILE;
END //

-- 2. Sum of N numbers
CREATE PROCEDURE GetSumOfN(IN n INT, OUT total INT)
BEGIN
    SET total = (n * (n + 1)) / 2;
END //

-- 3. Leap Year
CREATE PROCEDURE CheckLeapYear(IN yr INT)
BEGIN
    IF (yr % 4 = 0 AND yr % 100 != 0) OR (yr % 400 = 0) THEN
        SELECT 'Leap Year' AS Result;
    ELSE
        SELECT 'Not Leap Year' AS Result;
    END IF;
END //

-- 5. Areas Table and Process
CREATE TABLE Areas (Radius INT, Area DECIMAL(10,2));
CREATE PROCEDURE CalculateAreas()
BEGIN
    DECLARE r INT DEFAULT 6;
    WHILE r <= 10 DO
        INSERT INTO Areas (Radius, Area) VALUES (r, PI() * r * r);
        SET r = r + 1;
    END WHILE;
END //

-- 6. Add 1000 to bal_due (Max 6000 check)
CREATE PROCEDURE UpdateClientBal(IN p_client_no VARCHAR(6))
BEGIN
    DECLARE current_bal DECIMAL(10,2);
    SELECT Bal_due INTO current_bal FROM client_master WHERE Client_no = p_client_no;
    IF current_bal < 6000 THEN
        UPDATE client_master SET Bal_due = Bal_due + 1000 WHERE Client_no = p_client_no;
    END IF;
END //

Assignment 8: Prime Entries and Triggers

CREATE TABLE Prime_Entry (
    Num_id INT AUTO_INCREMENT PRIMARY KEY,
    Prime_num INT NOT NULL UNIQUE
);

DELIMITER //
CREATE FUNCTION IsPrime(p_num INT) RETURNS BOOLEAN DETERMINISTIC
BEGIN
    DECLARE i INT DEFAULT 2;
    IF p_num <= 1 THEN RETURN FALSE; END IF;
    WHILE i <= p_num / 2 DO
        IF p_num % i = 0 THEN RETURN FALSE; END IF;
        SET i = i + 1;
    END WHILE;
    RETURN TRUE;
END //

CREATE PROCEDURE AddPrime(IN p_num INT)
BEGIN
    DECLARE EXIT HANDLER FOR 1062 SELECT 'Number already exists in database' AS Error;
    IF IsPrime(p_num) THEN
        INSERT INTO Prime_Entry (Prime_num) VALUES (p_num);
        SELECT 'Prime Inserted' AS Message;
    ELSE
        SELECT 'Not a Prime' AS Message;
    END IF;
END //
DELIMITER ;

DELIMITER //
-- 1. Withdraw simulation with Undo/Savepoint
CREATE PROCEDURE AdvancedWithdraw(IN p_acc_no VARCHAR(8))
BEGIN
    DECLARE v_total DECIMAL(12,4);
    START TRANSACTION;
    UPDATE Acc_details SET Total_amt = Total_amt - 500 WHERE Acc_no = p_acc_no;
    SAVEPOINT sp1;
    UPDATE Acc_details SET Total_amt = Total_amt - 500 WHERE Acc_no = p_acc_no;
    SELECT Total_amt INTO v_total FROM Acc_details WHERE Acc_no = p_acc_no;
    IF v_total < 1000 THEN
        ROLLBACK TO sp1;
        SELECT 'Undoing last transaction. Balance below 1000' AS Status;
    END IF;
    COMMIT;
END //

-- 5. Emp Salary Increment Process
CREATE PROCEDURE ProcessResearchIncrements()
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE v_id INT;
    DECLARE v_sal DECIMAL(7,2);
    DECLARE cur CURSOR FOR SELECT empno, sal FROM emp WHERE deptno = 20; -- RESEARCH is dept 20
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;
    
    CREATE TABLE IF NOT EXISTS Emp_sal_inc (
        Emp_id INT, Cur_sal DECIMAL(7,2), Inc_date DATE
    );

    OPEN cur;
    read_loop: LOOP
        FETCH cur INTO v_id, v_sal;
        IF done THEN LEAVE read_loop; END IF;
        
        UPDATE emp SET sal = sal * 1.02 WHERE empno = v_id;
        INSERT INTO Emp_sal_inc VALUES (v_id, v_sal * 1.02, CURDATE());
    END LOOP;
    CLOSE cur;
END //
DELIMITER ;

DELIMITER //
-- 1. Apply 2% interest using Cursors & Functions
CREATE FUNCTION Gen_Trans_ID() RETURNS INT DETERMINISTIC
BEGIN
    DECLARE next_id INT;
    SELECT IFNULL(MAX(Transaction_id), 0) + 1 INTO next_id FROM Transactions_Acc;
    RETURN next_id;
END //

CREATE PROCEDURE AddInterest()
BEGIN
    DECLARE done INT DEFAULT FALSE;
    DECLARE v_acc VARCHAR(8);
    DECLARE v_amt DECIMAL(12,4);
    DECLARE v_interest DECIMAL(12,4);
    DECLARE cur CURSOR FOR SELECT Acc_no, Total_amt FROM Acc_details WHERE Acc_status = 'A';
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;

    OPEN cur;
    read_loop: LOOP
        FETCH cur INTO v_acc, v_amt;
        IF done THEN LEAVE read_loop; END IF;
        
        SET v_interest = v_amt * 0.02;
        UPDATE Acc_details SET Total_amt = Total_amt + v_interest WHERE Acc_no = v_acc;
        
        INSERT INTO Transactions_Acc (Transaction_id, Acc_no, Deposit_amt, Withdraw_amt, Mode_trans, Trans_date)
        VALUES (Gen_Trans_ID(), v_acc, v_interest, 0, 'CASH', CURDATE());
    END LOOP;
    CLOSE cur;
END //
DELIMITER ;

-- 2. Audit Table & Triggers
CREATE TABLE Emp_audit (
    Emp_no DECIMAL(4) PRIMARY KEY,
    Dept_no DECIMAL(4) NOT NULL,
    Status VARCHAR(8),
    Salary DECIMAL(8,2) NOT NULL,
    Audit_date DATE NOT NULL,
    FOREIGN KEY (Dept_no) REFERENCES dept(deptno)
);

DELIMITER //
CREATE TRIGGER trg_audit_update
AFTER UPDATE ON emp FOR EACH ROW
BEGIN
    INSERT IGNORE INTO Emp_audit (Emp_no, Dept_no, Status, Salary, Audit_date)
    VALUES (OLD.empno, OLD.deptno, 'UPDATED', OLD.sal, CURDATE());
END //

CREATE TRIGGER trg_audit_delete
AFTER DELETE ON emp FOR EACH ROW
BEGIN
    INSERT IGNORE INTO Emp_audit (Emp_no, Dept_no, Status, Salary, Audit_date)
    VALUES (OLD.empno, OLD.deptno, 'DELETED', OLD.sal, CURDATE());
END //
DELIMITER ;
